CREATE TABLE public.daily_products (
id uuid NOT NULL DEFAULT gen_random_uuid(),
basepaint_day integer NOT NULL UNIQUE,
theme text,
art_url text NOT NULL,
art_storage_path text,
palette_json jsonb,
palette_color text,
shirt_color text NOT NULL,
front_print_url text NOT NULL,
back_print_url text NOT NULL,
mockup_front_url text,
mockup_back_url text,
mockups_json jsonb,
printify_product_id text,
printify_blueprint_id integer,
printify_print_provider_id integer,
printify_variants_json jsonb,
printify_sync_status text,
printify_synced_at timestamp with time zone,
price_cents integer NOT NULL,
currency text NOT NULL DEFAULT 'usd'::text,
status text NOT NULL DEFAULT 'active'::text,
created_at timestamp with time zone DEFAULT now(),
updated_at timestamp with time zone DEFAULT now(),
CONSTRAINT daily_products_pkey PRIMARY KEY (id)
);
CREATE TABLE public.orders (
id uuid NOT NULL DEFAULT gen_random_uuid(),
daily_product_id uuid,
stripe_session_id text UNIQUE,
stripe_payment_intent_id text,
fulfillment_order_id text,
customer_email text,
customer_name text,
shipping_json jsonb,
size text NOT NULL,
status text NOT NULL DEFAULT 'paid'::text,
fulfillment_mode text NOT NULL,
fulfillment_provider text NOT NULL DEFAULT 'printify'::text,
fulfillment_payload jsonb,
shipping_cost_cents integer,
total_price_cents integer,
printify_variant_id integer,
printify_blueprint_id integer,
printify_print_provider_id integer,
printify_shipping_method integer DEFAULT 1,
payment_reference text UNIQUE,
payment_method text,
payment_status text NOT NULL DEFAULT 'pending'::text,
chain_id integer,
receiver_address text,
expected_amount_wei text,
display_amount_eth text,
eth_usd_price text,
received_amount_wei text,
payer_address text,
payment_tx_hash text UNIQUE,
paid_at timestamp with time zone,
expires_at timestamp with time zone,
fulfillment_error text,
created_at timestamp with time zone DEFAULT now(),
updated_at timestamp with time zone DEFAULT now(),
CONSTRAINT orders_pkey PRIMARY KEY (id),
CONSTRAINT orders_daily_product_id_fkey FOREIGN KEY (daily_product_id) REFERENCES public.daily_products(id)
);
CREATE TABLE public.payment_events (
id uuid NOT NULL DEFAULT gen_random_uuid(),
tx_hash text NOT NULL UNIQUE,
from_address text,
to_address text,
amount_wei text,
matched_order_id uuid,
source text NOT NULL,
raw_payload jsonb,
created_at timestamp with time zone DEFAULT now(),
CONSTRAINT payment_events_pkey PRIMARY KEY (id),
CONSTRAINT payment_events_matched_order_id_fkey FOREIGN KEY (matched_order_id) REFERENCES public.orders(id)
);
CREATE TABLE public.agent_runs (
id uuid NOT NULL DEFAULT gen_random_uuid(),
basepaint_day integer,
status text NOT NULL,
message text,
logs jsonb,
created_at timestamp with time zone DEFAULT now(),
CONSTRAINT agent_runs_pkey PRIMARY KEY (id)
);
